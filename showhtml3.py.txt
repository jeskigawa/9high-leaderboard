#!/usr/bin/env python3
"""rangeEnd入力欄とuseState宣言を確認"""
HTML = '/var/www/leaderboard/index.html'
with open(HTML, encoding='utf-8') as f:
    lines = f.readlines()

# 1) 行2833-2870 (rangeEnd input周辺)
print("=== 行2833-2870 ===")
for i in range(2832, min(2870, len(lines))):
    print(f"{i+1:5}: {lines[i].rstrip()}")

# 2) rangeEnd / rangeStart の useState 宣言行
print("\n=== useState rangeStart/rangeEnd ===")
for i, line in enumerate(lines):
    ls = line.rstrip()
    if ('rangeStart' in ls or 'rangeEnd' in ls) and 'useState' in ls:
        print(f"{i+1:5}: {ls[:120]}")
