#!/usr/bin/env python3
"""Phase1のページループ停止条件を確認"""
APP = '/var/www/leaderboard/app.py'
with open(APP, encoding='utf-8') as f:
    lines = f.readlines()

# '対象イベント' を含む行とその前後を表示
print("=== 対象イベントなし の行とコンテキスト ===")
shown = set()
for i, line in enumerate(lines):
    if i >= 2714:
        break
    if '対象イベント' in line or 'skip' in line.lower():
        start = max(0, i-6)
        end = min(2714, i+4)
        if start not in shown:
            for j in range(start, end):
                shown.add(j)
                print(f"{j+1:5}: {lines[j].rstrip()}")
            print("---")

# Phase1のページループを探す
print("\n=== Phase1 while/for ページループ ===")
for i, line in enumerate(lines):
    if i >= 2714:
        break
    ls = line.rstrip()
    if ('while True' in ls or 'while true' in ls.lower()) and i > 2500:
        for j in range(i, min(i+30, 2714)):
            print(f"{j+1:5}: {lines[j].rstrip()}")
        print("---")
        break

# eff_end / eff_start 比較ロジックを探す
print("\n=== eff_end / eff_start 比較 (Phase1内) ===")
for i, line in enumerate(lines):
    if i >= 2714:
        break
    if ('eff_end' in line or 'eff_start' in line) and ('>' in line or '<' in line):
        print(f"{i+1:5}: {lines[i].rstrip()}")
