#!/usr/bin/env python3
"""pokerfans関連のルートとファイルを確認"""
import os, re

APP = '/var/www/leaderboard/app.py'
with open(APP, encoding='utf-8') as f:
    lines = f.readlines()

print("=== pokerfans関連ルート ===")
for i, line in enumerate(lines, 1):
    if 'pokerfans' in line.lower():
        print(f"行{i}: {line.rstrip()}")

print("\n=== /var/www/leaderboard/ のhtmlファイル ===")
for fn in os.listdir('/var/www/leaderboard'):
    if fn.endswith('.html'):
        print(f"  {fn}")
