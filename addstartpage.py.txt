#!/usr/bin/env python3
"""
Sunvy取込に「開始ページ」手動指定機能を追加
- app.py: startPage パラメータを受け取り Phase1 ループで優先使用
- index.html: 開始ページ入力欄を追加
"""
import subprocess, sys, re, tempfile, os

APP  = '/var/www/leaderboard/app.py'
HTML = '/var/www/leaderboard/index.html'

subprocess.run(['cp', APP,  '/tmp/app.py.bak.startpage'])
subprocess.run(['cp', HTML, '/tmp/index.html.bak.startpage'])
print('バックアップ完了')

# ════════════════════════════════════════════
#  app.py 修正
# ════════════════════════════════════════════
with open(APP, encoding='utf-8') as f:
    ac = f.read()

# 1) end_date 行を見つけてその後に startPage 受け取りを追加
m = re.search(r'^( *)end_date\s*=\s*req\.get\(.*?endDate.*?\n', ac, re.MULTILINE)
if not m:
    print('ERROR app.py: end_date 行が見つかりません')
    sys.exit(1)
indent = m.group(1)
old_end = m.group(0)
new_end = (
    old_end +
    f'{indent}try:\n'
    f'{indent}    _req_start_page = int(req.get(\'startPage\', -1))\n'
    f'{indent}except (ValueError, TypeError):\n'
    f'{indent}    _req_start_page = -1\n'
)
ac = ac.replace(old_end, new_end, 1)
print('OK app.py: startPage受け取り追加')

# 2) _start_page 計算に手動指定を優先させる
m2 = re.search(
    r'([ \t]*)# 期間指定の場合、日付から開始ページを計算[^\n]*\n'
    r'\1if mode == .range. and eff_end:',
    ac
)
if not m2:
    print('ERROR app.py: _start_page ループパターンが見つかりません')
    sys.exit(1)
lind = m2.group(1)
old_loop = m2.group(0)
new_loop = (
    f'{lind}# 期間指定の場合、日付から開始ページを計算（1ページ約2日）\n'
    f'{lind}if _req_start_page >= 0:\n'
    f'{lind}    _start_page = _req_start_page\n'
    f"{lind}elif mode == 'range' and eff_end:"
)
ac = ac.replace(old_loop, new_loop, 1)
print('OK app.py: 手動startPage優先ロジック追加')

# 構文チェック
with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False, encoding='utf-8') as tmp:
    tmp.write(ac)
    tp = tmp.name
r = subprocess.run(['python3', '-m', 'py_compile', tp], capture_output=True, text=True)
os.unlink(tp)
if r.returncode != 0:
    print(f'ERROR app.py 構文エラー!\n{r.stderr}')
    sys.exit(1)
print('OK app.py: 構文チェック通過')

# ════════════════════════════════════════════
#  index.html 修正
# ════════════════════════════════════════════
with open(HTML, encoding='utf-8') as f:
    hc = f.read()

# 1) useState に startPageNum を追加
m3 = re.search(r"([ \t]*const \[rangeEnd, setRangeEnd\] = React\.useState\(''\);)", hc)
if not m3:
    print('ERROR index.html: rangeEnd useState が見つかりません')
    sys.exit(1)
old_state = m3.group(0)
sind = m3.group(1).split('const')[0]  # インデント取得
new_state = old_state + "\n" + sind + "const [startPageNum, setStartPageNum] = React.useState('');"
hc = hc.replace(old_state, new_state, 1)
print('OK index.html: startPageNum state追加')

# 2) fetch body に startPage を追加
m4 = re.search(r"([ \t]*endDate: importMode === 'range' \? rangeEnd : undefined,)", hc)
if not m4:
    print('ERROR index.html: fetch body endDate が見つかりません')
    sys.exit(1)
old_body = m4.group(0)
bind = m4.group(1).split('endDate')[0]
new_body = (
    old_body + "\n" +
    bind + "startPage: importMode === 'range' && startPageNum !== '' "
    "? parseInt(startPageNum) : undefined,"
)
hc = hc.replace(old_body, new_body, 1)
print('OK index.html: fetch body startPage追加')

# 3) フォームに開始ページ入力欄を追加（rangeEnd入力の直後の </div> の後）
m5 = re.search(
    r'(onChange=\{e => setRangeEnd\(e\.target\.value\)\} />\s*\n([ \t]*)</div>)',
    hc
)
if not m5:
    print('ERROR index.html: rangeEnd input が見つかりません')
    sys.exit(1)
old_form = m5.group(0)
fi = m5.group(2)  # </div>のインデント
new_form = (
    old_form +
    f'\n{fi}<div className="form-group" style={{{{ flex: \'0 0 110px\' }}}}>\n'
    f'{fi}  <label className="form-label">開始ページ（省略可）</label>\n'
    f'{fi}  <input\n'
    f'{fi}    className="form-input"\n'
    f'{fi}    type="number"\n'
    f'{fi}    min="0"\n'
    f'{fi}    placeholder="自動"\n'
    f'{fi}    value={{startPageNum}}\n'
    f'{fi}    onChange={{e => setStartPageNum(e.target.value)}}\n'
    f'{fi}  />\n'
    f'{fi}</div>'
)
hc = hc.replace(old_form, new_form, 1)
print('OK index.html: 開始ページ入力欄追加')

# ════════════════════════════════════════════
#  書き込み＆再起動
# ════════════════════════════════════════════
with open(APP, 'w', encoding='utf-8') as f:
    f.write(ac)
print('OK: app.py 書き込み完了')

with open(HTML, 'w', encoding='utf-8') as f:
    f.write(hc)
print('OK: index.html 書き込み完了')

subprocess.run(['systemctl', 'restart', 'leaderboard'])
import time; time.sleep(2)
r = subprocess.run(['systemctl', 'is-active', 'leaderboard'], capture_output=True, text=True)
status = r.stdout.strip()
print(f'leaderboard: {status}')

if status != 'active':
    print('WARNING: 起動失敗。バックアップに戻します...')
    subprocess.run(['cp', '/tmp/app.py.bak.startpage',   APP])
    subprocess.run(['cp', '/tmp/index.html.bak.startpage', HTML])
    subprocess.run(['systemctl', 'restart', 'leaderboard'])
    sys.exit(1)

print('\n=== 完了 ===')
print('取込画面に「開始ページ（省略可）」入力欄が追加されました')
print('空白 or 0 のまま → 自動計算')
print('数字を入力 → そのページから開始')
