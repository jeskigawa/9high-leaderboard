#!/usr/bin/env python3
"""開始ページの1始まり→0始まり変換を修正"""
import subprocess, sys, re, tempfile, os

APP = '/var/www/leaderboard/app.py'
subprocess.run(['cp', APP, '/tmp/app.py.bak.spfix'])

with open(APP, encoding='utf-8') as f:
    ac = f.read()

# 現在: int(req.get('startPage', -1)) → ユーザー入力をそのまま使用（0始まり）
# 修正: ユーザーは1始まりで入力するので -1 して0始まりに変換
OLD = (
    "        try:\n"
    "            _req_start_page = int(req.get('startPage', -1))\n"
    "        except (ValueError, TypeError):\n"
    "            _req_start_page = -1"
)
NEW = (
    "        try:\n"
    "            _sp_val = req.get('startPage', None)\n"
    "            if _sp_val is not None and str(_sp_val).strip() != '':\n"
    "                _req_start_page = max(0, int(_sp_val) - 1)  # 1始まり→0始まり\n"
    "            else:\n"
    "                _req_start_page = -1\n"
    "        except (ValueError, TypeError):\n"
    "            _req_start_page = -1"
)

if OLD in ac:
    ac = ac.replace(OLD, NEW, 1)
    print('OK: 1始まり→0始まり変換を追加')
else:
    print('ERROR: パターンが見つかりません')
    sys.exit(1)

# 構文チェック
with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False, encoding='utf-8') as tmp:
    tmp.write(ac)
    tp = tmp.name
r = subprocess.run(['python3', '-m', 'py_compile', tp], capture_output=True, text=True)
os.unlink(tp)
if r.returncode != 0:
    print(f'ERROR: 構文エラー!\n{r.stderr}')
    sys.exit(1)
print('OK: 構文チェック通過')

with open(APP, 'w', encoding='utf-8') as f:
    f.write(ac)

subprocess.run(['systemctl', 'restart', 'leaderboard'])
import time; time.sleep(2)
r = subprocess.run(['systemctl', 'is-active', 'leaderboard'], capture_output=True, text=True)
print(f'leaderboard: {r.stdout.strip()}')
print('完了: 開始ページ16と入力→ページ16からスキャン開始')
