#!/usr/bin/env python3
"""行2517-2560を表示"""
APP = '/var/www/leaderboard/app.py'
with open(APP, encoding='utf-8') as f:
    lines = f.readlines()

print("=== 行2517-2560 ===")
for i in range(2516, min(2560, len(lines))):
    print(f"{i+1:5}: {lines[i].rstrip()}")

# JS: fetchやXHRでimport-eventsを呼んでいる行を幅広く探す
print("\n=== JS: fetch/XHR/EventSource (sunvy or import) ===")
for i, line in enumerate(lines):
    ls = line.rstrip()
    if i < 2400:
        continue
    if ('fetch(' in ls or 'EventSource' in ls or 'XMLHttpRequest' in ls) and \
       ('sunvy' in ls.lower() or 'import' in ls.lower()):
        for j in range(max(0,i-2), min(len(lines),i+5)):
            print(f"{j+1:5}: {lines[j].rstrip()}")
        print("---")
