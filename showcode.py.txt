#!/usr/bin/env python3
"""
app.py のログ関数とPhase2コードを確認する
"""
APP = '/var/www/leaderboard/app.py'

with open(APP, encoding='utf-8') as f:
    lines = f.readlines()

code = ''.join(lines)

# ── 1. _append_import_log 関数を探す ─────────────────────────────
print("=== 1. _append_import_log 関数 ===")
for i, line in enumerate(lines):
    if '_append_import_log' in line and 'def ' in line:
        # 関数定義から20行表示
        for j in range(i, min(i+20, len(lines))):
            print(f"{j+1:5}: {lines[j]}", end='')
        break
else:
    # 関数定義が見つからなければ、最初の使用箇所を探す
    print("def が見つかりません。lambdaか別の形式を確認:")
    for i, line in enumerate(lines):
        if '_append_import_log' in line:
            print(f"{i+1:5}: {line}", end='')
            if i > 10: break  # 最初の10件のみ

# ── 2. SUNVY_IMPORT_LOG グローバル変数を探す ─────────────────────
print("\n=== 2. import_log / SUNVY_IMPORT_LOG 変数 ===")
for i, line in enumerate(lines):
    lstrip = line.strip()
    if ('import_log' in lstrip or 'IMPORT_LOG' in lstrip) and i < 200:
        print(f"{i+1:5}: {line}", end='')

# ── 3. Phase2スレッド開始部分 ────────────────────────────────────
print("\n=== 3. Phase2スレッド開始確認 ===")
for i, line in enumerate(lines):
    if 'phase2_done = threading.Event()' in line:
        print(f"Phase2スレッドブロック開始: 行{i+1}")
        for j in range(i, min(i+10, len(lines))):
            print(f"{j+1:5}: {lines[j]}", end='')
        break
else:
    print("WARNING: threading.Event() が見つかりません！fixphase2thread.py未適用の可能性")

# ── 4. generate() のfinally ──────────────────────────────────────
print("\n=== 4. generate() のfinallyブロック ===")
in_gen = False
for i, line in enumerate(lines):
    if 'def generate():' in line:
        in_gen = True
    if in_gen and '        finally:' in line:
        for j in range(i, min(i+6, len(lines))):
            print(f"{j+1:5}: {lines[j]}", end='')
        break

# ── 5. sunvy_import_log ファイルがあるか確認 ─────────────────────
print("\n=== 5. ログファイル確認 ===")
import os, glob
patterns = [
    '/var/www/leaderboard/data/*.log',
    '/var/www/leaderboard/*.log',
    '/var/log/leaderboard*',
    '/tmp/leaderboard*',
    '/tmp/sunvy*',
]
found = False
for pat in patterns:
    files = glob.glob(pat)
    for f in files:
        print(f"発見: {f} ({os.path.getsize(f)} bytes)")
        found = True
if not found:
    print("ログファイルは見つかりません（インメモリ管理）")

# ── 6. 最近更新されたファイル ──────────────────────────────────────
print("\n=== 6. /var/www/leaderboard/data/ の最新ファイル ===")
import time
data_dir = '/var/www/leaderboard/data'
if os.path.isdir(data_dir):
    files = [(os.path.getmtime(os.path.join(data_dir,f)), f)
             for f in os.listdir(data_dir)]
    for mtime, fname in sorted(files, reverse=True)[:10]:
        print(f"  {time.strftime('%Y-%m-%d %H:%M:%S', time.localtime(mtime))}  {fname}")
