#!/usr/bin/env python3
"""index.htmlのSunvy取込フォームとfetch呼び出しを確認"""
HTML = '/var/www/leaderboard/index.html'
with open(HTML, encoding='utf-8') as f:
    lines = f.readlines()

print(f"総行数: {len(lines)}")

# 1) 行2355-2395 (fetch呼び出し周辺)
print("\n=== 行2355-2395: fetch呼び出し ===")
for i in range(2354, min(2395, len(lines))):
    print(f"{i+1:5}: {lines[i].rstrip()}")

# 2) rangeStart / rangeEnd が定義・使用されている行
print("\n=== rangeStart / rangeEnd 使用箇所 ===")
for i, line in enumerate(lines):
    ls = line.rstrip()
    if 'rangeStart' in ls or 'rangeEnd' in ls:
        print(f"{i+1:5}: {ls[:120]}")

# 3) Sunvy import モーダルのHTML (input type=date付近)
print("\n=== type=date 入力欄 ===")
for i, line in enumerate(lines):
    ls = line.rstrip()
    if 'type="date"' in ls or "type='date'" in ls:
        for j in range(max(0,i-3), min(len(lines),i+4)):
            print(f"{j+1:5}: {lines[j].rstrip()}")
        print("---")
