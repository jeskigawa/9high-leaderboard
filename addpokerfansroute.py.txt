#!/usr/bin/env python3
"""app.py に /pokerfans ルートを追加"""
import subprocess, sys, re, tempfile, os

APP = '/var/www/leaderboard/app.py'
subprocess.run(['cp', APP, '/tmp/app.py.bak.pokerfans'])

with open(APP, encoding='utf-8') as f:
    ac = f.read()

# すでに /pokerfans ルートがあれば何もしない
if "'/pokerfans'" in ac or '"/pokerfans"' in ac:
    print('すでに /pokerfans ルートが存在します')
    sys.exit(0)

# send_from_directory が import されているか確認
if 'send_from_directory' not in ac:
    # flask import 行に追加
    ac = re.sub(
        r'(from flask import[^\n]+)',
        lambda m: m.group(0) if 'send_from_directory' in m.group(0)
                  else m.group(0).rstrip() + ', send_from_directory',
        ac, count=1
    )
    print('OK: send_from_directory を import に追加')

# app = Flask(__name__) の後にルートを追加
# または既存ルートの前に追加（@app.route('/' の最初の定義の前）
route_code = """
@app.route('/pokerfans')
def pokerfans_page():
    return send_from_directory('/var/www/leaderboard', 'pokerfans.html')

"""

# @app.route('/') を探して直前に挿入
m = re.search(r"(@app\.route\('/')", ac)
if m:
    ac = ac[:m.start()] + route_code + ac[m.start():]
    print('OK: /pokerfans ルートを追加')
else:
    # if __name__ == の前に追加
    m2 = re.search(r"if __name__\s*==\s*['\"]__main__['\"]", ac)
    if m2:
        ac = ac[:m2.start()] + route_code + ac[m2.start():]
        print('OK: /pokerfans ルートを追加（末尾）')
    else:
        print('ERROR: 挿入箇所が見つかりません')
        sys.exit(1)

# 構文チェック
with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False, encoding='utf-8') as tmp:
    tmp.write(ac)
    tp = tmp.name
r = subprocess.run(['python3', '-m', 'py_compile', tp], capture_output=True, text=True)
os.unlink(tp)
if r.returncode != 0:
    print(f'ERROR: 構文エラー!\n{r.stderr}')
    sys.exit(1)
print('OK: 構文チェック通過')

with open(APP, 'w', encoding='utf-8') as f:
    f.write(ac)

subprocess.run(['systemctl', 'restart', 'leaderboard'])
import time; time.sleep(2)
r = subprocess.run(['systemctl', 'is-active', 'leaderboard'], capture_output=True, text=True)
print(f'leaderboard: {r.stdout.strip()}')
print('完了: https://9high.net/pokerfans でアクセス可能になりました')
