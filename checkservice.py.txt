import subprocess

r1 = subprocess.run(['systemctl','status','leaderboard','--no-pager'],
                    capture_output=True, text=True)
print(r1.stdout[-1000:])

r2 = subprocess.run(['journalctl','-u','leaderboard','-n','30','--no-pager'],
                    capture_output=True, text=True)
print(r2.stdout[-2000:])
