import subprocess

path = '/var/www/leaderboard/app.py'
lines = open(path).readlines()
before = len(lines)

new_lines = [l for l in lines
             if l.strip() != '@login_required'
             and l.strip() != '@admin_required']

removed = before - len(new_lines)
print('removed:', removed, 'decorators')

open(path, 'w').writelines(new_lines)

r = subprocess.run(['systemctl','restart','leaderboard'],capture_output=True,text=True)
import time; time.sleep(4)
r2 = subprocess.run(['systemctl','is-active','leaderboard'],capture_output=True,text=True)
print('service:', r2.stdout.strip())
