import subprocess

link = '  <link rel="icon" href="/leaderboard/static/favicon.svg" type="image/svg+xml">\n'

for fname in ['9high-leaderboard.html', 'index.html', 'admin.html']:
    path = '/var/www/leaderboard/' + fname
    try:
        lines = open(path).readlines()
    except FileNotFoundError:
        print('skip (not found):', fname)
        continue
    if any('rel="icon"' in l for l in lines):
        print('already has icon:', fname)
        continue
    new = []
    for l in lines:
        new.append(l)
        if '<head>' in l:
            new.append(link)
    open(path, 'w').writelines(new)
    print('updated:', fname)

r = subprocess.run(['systemctl','restart','leaderboard'],capture_output=True,text=True)
import time; time.sleep(4)
r2 = subprocess.run(['systemctl','is-active','leaderboard'],capture_output=True,text=True)
print('service:', r2.stdout.strip())
