#!/usr/bin/env python3
"""開始ページの1始まり→0始まり変換を修正（regex版）"""
import subprocess, sys, re, tempfile, os

APP = '/var/www/leaderboard/app.py'
subprocess.run(['cp', APP, '/tmp/app.py.bak.spfix2'])

with open(APP, encoding='utf-8') as f:
    ac = f.read()

# _req_start_page = int(req.get('startPage', -1)) の行を探してその行だけ書き換え
m = re.search(
    r'([ \t]*)(_req_start_page\s*=\s*int\(req\.get\([\'"]startPage[\'"],\s*-1\)\))',
    ac
)
if not m:
    print('ERROR: _req_start_page 行が見つかりません。現在のstartPage関連行:')
    for i, line in enumerate(ac.splitlines(), 1):
        if 'startPage' in line or '_req_start_page' in line:
            print(f'  行{i}: {line}')
    sys.exit(1)

ind = m.group(1)
old_line = m.group(0)
# ユーザーは1始まりで入力→内部は0始まり
new_block = (
    f"{ind}_sp_raw = req.get('startPage', None)\n"
    f"{ind}try:\n"
    f"{ind}    if _sp_raw is not None and str(_sp_raw).strip() != '' and int(_sp_raw) >= 1:\n"
    f"{ind}        _req_start_page = int(_sp_raw) - 1  # 1始まり→0始まり変換\n"
    f"{ind}    else:\n"
    f"{ind}        _req_start_page = -1\n"
    f"{ind}except (ValueError, TypeError):\n"
    f"{ind}    _req_start_page = -1"
)
ac = ac.replace(old_line, new_block, 1)
print('OK: _req_start_page の1始まり変換を追加')

# 構文チェック
with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False, encoding='utf-8') as tmp:
    tmp.write(ac)
    tp = tmp.name
r = subprocess.run(['python3', '-m', 'py_compile', tp], capture_output=True, text=True)
os.unlink(tp)
if r.returncode != 0:
    print(f'ERROR: 構文エラー!\n{r.stderr}')
    sys.exit(1)
print('OK: 構文チェック通過')

with open(APP, 'w', encoding='utf-8') as f:
    f.write(ac)

subprocess.run(['systemctl', 'restart', 'leaderboard'])
import time; time.sleep(2)
r = subprocess.run(['systemctl', 'is-active', 'leaderboard'], capture_output=True, text=True)
print(f'leaderboard: {r.stdout.strip()}')
print('完了: 16入力→page=15（16ページ目）からスキャン')
