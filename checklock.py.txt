#!/usr/bin/env python3
"""ロック名とPhase2スレッドの有無を確認"""
APP = '/var/www/leaderboard/app.py'
with open(APP, encoding='utf-8') as f:
    lines = f.readlines()

print("=== LOCK関連行 ===")
for i, line in enumerate(lines):
    if 'LOCK' in line and ('Lock()' in line or '.acquire' in line or '.release' in line):
        print(f"{i+1:5}: {line.rstrip()}")

print("\n=== phase2_done / threading.Event の有無 ===")
found = False
for i, line in enumerate(lines):
    if 'threading.Event' in line or 'phase2_done' in line or '_run_phase2' in line:
        print(f"{i+1:5}: {line.rstrip()}")
        found = True
if not found:
    print("見つかりません → fixphase2thread.py 未適用")

print("\n=== Phase2開始マーカー周辺 ===")
for i, line in enumerate(lines):
    if '# --- Phase 2' in line:
        for j in range(i, min(i+8, len(lines))):
            print(f"{j+1:5}: {lines[j].rstrip()}")
        break
