#!/usr/bin/env python3
"""5月分イベントの有無とイベントのキー構造を確認"""
import json, os

DATA = '/var/www/leaderboard/data/sunvy_imported_events.json'
with open(DATA, encoding='utf-8') as f:
    data = json.load(f)

events = data.get('events', [])
print(f"取込済みイベント総数: {len(events)} 件")

# 5月のイベントを探す
may_events = [e for e in events if str(e.get('date','') or '').startswith('2026-05')]
print(f"2026年5月のイベント: {len(may_events)} 件")

# 最新の5月イベントを表示
if may_events:
    print("\n=== 5月イベント（最新3件）===")
    for ev in may_events[-3:]:
        print(json.dumps(ev, ensure_ascii=False, indent=2)[:500])
        print("---")
else:
    print("\n5月のイベントは見つかりません")

# イベントの全キー一覧（最初の1件から）
if events:
    print("\n=== イベントのキー構造（1件目） ===")
    print(list(events[0].keys()))
    print(json.dumps(events[0], ensure_ascii=False, indent=2)[:400])
