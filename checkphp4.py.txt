import subprocess

# アーキテクチャ確認
r = subprocess.run('dpkg --print-architecture', shell=True, capture_output=True, text=True)
print('arch:', r.stdout.strip())

# ondrej のキャッシュファイル確認
r2 = subprocess.run('ls -lh /var/lib/apt/lists/ | grep ondrej',
                    shell=True, capture_output=True, text=True)
print('=== ondrej list files ===')
print(r2.stdout or 'ファイルなし')

# キャッシュをクリアして再取得
print('=== apt cache クリア & 再取得 ===')
subprocess.run('rm -f /var/lib/apt/lists/ppa.launchpad.net*',
               shell=True)
r3 = subprocess.run('apt-get update 2>&1 | grep -E "ondrej|ERR|Err"',
                    shell=True, capture_output=True, text=True)
print(r3.stdout or 'エラーなし')

# 再確認
r4 = subprocess.run('apt-cache show php8.2-fpm 2>&1 | head -5',
                    shell=True, capture_output=True, text=True)
print('=== php8.2-fpm ===')
print(r4.stdout or r4.stderr)
