#!/usr/bin/env python3
"""Sunvy取込UIの詳細確認 - 変数名を広く検索"""
APP = '/var/www/leaderboard/app.py'
with open(APP, encoding='utf-8') as f:
    lines = f.readlines()

# 1) text/event-stream が出てくる行（SSEエンドポイント一覧）
print("=== SSE (text/event-stream) エンドポイント ===")
for i, line in enumerate(lines):
    if 'event-stream' in line or 'stream_with_context' in line:
        print(f"{i+1:5}: {line.rstrip()}")

# 2) eff_start / eff_end の代入行（request不問）
print("\n=== eff_start / eff_end 代入行 ===")
for i, line in enumerate(lines):
    ls = line.rstrip()
    if ('eff_start' in ls or 'eff_end' in ls) and '=' in ls and 'if' not in ls[:ls.find('=')]:
        print(f"{i+1:5}: {ls}")

# 3) HTMLの日付入力 (type="date" または date picker)
print("\n=== HTML: type=date 入力欄 ===")
for i, line in enumerate(lines):
    ls = line.rstrip()
    if 'type="date"' in ls or "type='date'" in ls or 'type=\\"date\\"' in ls:
        for j in range(max(0,i-2), min(len(lines),i+3)):
            print(f"{j+1:5}: {lines[j].rstrip()}")
        print("---")

# 4) JavaScriptでsunvyのURLを呼んでいる箇所
print("\n=== JS: sunvy import URL呼び出し ===")
for i, line in enumerate(lines):
    ls = line.rstrip()
    if ('EventSource' in ls or 'event-stream' in ls) and 'sunvy' in ls.lower():
        for j in range(max(0,i-3), min(len(lines),i+5)):
            print(f"{j+1:5}: {lines[j].rstrip()}")
        print("---")

# 5) sunvy関連のルート一覧
print("\n=== @app.route sunvy関連 ===")
for i, line in enumerate(lines):
    if '@app.route' in line and 'sunvy' in line.lower():
        print(f"{i+1:5}: {line.rstrip()}")
