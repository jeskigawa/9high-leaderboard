#!/usr/bin/env python3
"""
Phase1ページスキャン改善:
1. SUNVY_MAX_PAGES を 100 に増やす
2. 期間指定の場合、日付から開始ページを計算してスキップ
"""
import subprocess, sys, re

APP = '/var/www/leaderboard/app.py'
with open(APP, encoding='utf-8') as f:
    content = f.read()

subprocess.run(['cp', APP, '/tmp/app.py.bak.pagerange'])

changes = 0

# ── Step 1: SUNVY_MAX_PAGES を 100 に変更 ─────────────────────────
m = re.search(r'(SUNVY_MAX_PAGES\s*=\s*)(\d+)', content)
if m:
    old_val = m.group(2)
    content = re.sub(r'SUNVY_MAX_PAGES\s*=\s*\d+',
                     'SUNVY_MAX_PAGES = 100', content, count=1)
    print(f'OK: SUNVY_MAX_PAGES {old_val} -> 100')
    changes += 1
else:
    print('WARNING: SUNVY_MAX_PAGESが見つかりません')

# ── Step 2: ループを開始ページ計算版に変更 ────────────────────────
OLD_LOOP = '            for page in range(0, SUNVY_MAX_PAGES):'

# NOTE: \\n\\n はこのファイル内では \n\n としてapp.pyに書き込まれる（SSE終端）
NEW_LOOP = (
    '            # 期間指定の場合、日付から開始ページを計算（1ページ約2日）\n'
    "            if mode == 'range' and eff_end:\n"
    '                try:\n'
    '                    from datetime import date as _date\n'
    '                    _today = _date.today()\n'
    "                    _end_d = _date.fromisoformat(eff_end)\n"
    '                    _days_ago = (_today - _end_d).days\n'
    '                    _start_page = max(0, _days_ago // 2 - 5)\n'
    '                except Exception:\n'
    '                    _start_page = 0\n'
    '            else:\n'
    '                _start_page = 0\n'
    '            if _start_page > 0:\n'
    "                yield f\"data: {json.dumps({'type': 'status', 'message': f'ページ {_start_page + 1} からスキャン開始（{_start_page} ページスキップ）'})}\\n\\n\"\n"
    '            for page in range(_start_page, SUNVY_MAX_PAGES):'
)

if OLD_LOOP in content:
    content = content.replace(OLD_LOOP, NEW_LOOP, 1)
    print('OK: ループをスマート開始ページ版に変更')
    changes += 1
else:
    print('ERROR: ループパターンが見つかりません。現在の行:')
    for i, line in enumerate(content.splitlines(), 1):
        if 'for page in range' in line and 'MAX' in line:
            print(f'  行{i}: {line}')
    sys.exit(1)

if changes < 2:
    print('WARNING: 一部変更未適用')

# ── Step 3: 構文チェック ──────────────────────────────────────────
import tempfile, os
with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False, encoding='utf-8') as tmp:
    tmp.write(content)
    tmp_path = tmp.name

r = subprocess.run(['python3', '-m', 'py_compile', tmp_path], capture_output=True, text=True)
os.unlink(tmp_path)
if r.returncode != 0:
    print(f'ERROR: 構文エラー!\n{r.stderr}')
    sys.exit(1)
print('OK: 構文チェック通過')

# ── Step 4: 書き込み＆再起動 ──────────────────────────────────────
with open(APP, 'w', encoding='utf-8') as f:
    f.write(content)
print('OK: app.py を書き込みました')

subprocess.run(['systemctl', 'restart', 'leaderboard'])
import time; time.sleep(2)
r = subprocess.run(['systemctl', 'is-active', 'leaderboard'], capture_output=True, text=True)
status = r.stdout.strip()
print(f'leaderboard: {status}')

if status != 'active':
    print('WARNING: 起動失敗。バックアップに戻します...')
    subprocess.run(['cp', '/tmp/app.py.bak.pagerange', APP])
    subprocess.run(['systemctl', 'restart', 'leaderboard'])
    sys.exit(1)

print('')
print('=== 完了 ===')
print('次回の取込から:')
print('  - 最大100ページまでスキャン（従来: 5ページ）')
print('  - 期間指定時は日付から開始ページを自動計算')
