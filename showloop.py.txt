#!/usr/bin/env python3
"""Phase1ループ構造のみ抽出（1画面に収まる量）"""
APP = '/var/www/leaderboard/app.py'
with open(APP, encoding='utf-8') as f:
    lines = f.readlines()

print("=== for/while/page関連行 (行2500-2715) ===")
for i in range(2499, 2715):
    ls = lines[i].rstrip()
    # ループ制御・ページ変数に関係する行のみ表示
    if any(k in ls for k in [
        'for page', 'while ', 'page =', 'page+=', 'page +=',
        'max_page', 'MAX_PAGE', 'range(', 'page_num',
        'next_page', 'has_next', 'page_url', '?page=',
        'events/all', 'size=10'
    ]):
        print(f"{i+1:5}: {ls}")
